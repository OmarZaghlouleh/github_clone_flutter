class AuthRepoImp{

Future<Either<String, LoginResponse>> signUp({required SignUpParam signUpParam}) {
  return BaseApiClient.post<LoginResponse>(
      formData: FormData.fromMap(signUpParam.toJson()),
      url: Links.baseUrl + Links.startPoint + Links.register,
      converter: (value) {
        return LoginResponse.fromJson(value["data"]);
      });
}

}
